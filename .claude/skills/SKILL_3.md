```javascript
          <option>Fidelity IRA</option>
        </select>
      </div>
    `,
    footer: `
      <button class="btn btn-secondary" data-close-modal>Cancel</button>
      <button class="btn btn-primary" onclick="saveTransaction()">Save Transaction</button>
    `
  });
}

function saveTransaction() {
  const desc = document.getElementById('txDesc').value;
  const amount = parseFloat(document.getElementById('txAmount').value);
  if (!desc || !amount) {
    Toast.error('Missing fields', 'Please fill in description and amount');
    return;
  }
  // Save to localStorage / API here
  Toast.success('Transaction added', \`\${desc} — \${Fmt.currency(amount)}\`);
  document.querySelector('.modal-backdrop')?.remove();
  document.body.style.overflow = '';
}
```

## Data Persistence (localStorage pattern)

```javascript
// Simple localStorage store for FinanceTool data
const Store = {
  get(key) {
    try { return JSON.parse(localStorage.getItem(`ft_${key}`) || 'null'); }
    catch { return null; }
  },
  set(key, value) {
    localStorage.setItem(`ft_${key}`, JSON.stringify(value));
  },
  getTransactions() { return this.get('transactions') || []; },
  addTransaction(tx) {
    const txs = this.getTransactions();
    txs.unshift({ ...tx, id: Date.now(), date: tx.date || new Date().toISOString() });
    this.set('transactions', txs);
    return txs[0];
  },
  getBudgets() { return this.get('budgets') || {}; },
  getGoals() { return this.get('goals') || []; },
};
```
